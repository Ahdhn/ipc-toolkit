class HashItem {
public:
    EIGEN_MAKE_ALIGNED_OPERATOR_NEW

    int key;
    int id;
    AABB aabb;

    HashItem(int key, int id, const AABB& aabb)
        : key(key)
        , id(id)
        , aabb(aabb)
    {
    }

    bool operator<(const HashItem& other) const
    {
        if (key == other.key) {
            return id < other.id;
        }
        return key < other.key;
    }
};
